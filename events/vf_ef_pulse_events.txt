namespace = vfefpulse

country_event = {
	id = vfefpulse.1000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_federation = yes
		federation = { has_federation_type = vf_ef_federation }
		is_federation_leader = no
		is_subject = yes
		any_agreement = { agreement_preset = preset_auto_state }
	}

	immediate = {
		leave_alliance = { override_requirements = yes }
	}
}

country_event = { ##################################
	id = vfefpulse.2000 ##################################
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_federation = yes
		federation = { has_federation_type = vf_ef_federation }
		is_federation_leader = no
		check_subject_join_wars = yes
	}

	immediate = {
		random_war = {
			limit = {
				any_war_participant = {
					is_overlord_to = root
					is_war_leader = yes
				}	
			}
			every_war_participant = {
				limit = { is_same_empire = root }
				prev = { remove_war_participant = root }
			}
		}
	}
}

country_event = {
	id = vfefpulse.3000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_federation = yes
		federation = { has_federation_type = vf_ef_federation }
		# Changed from any_subject for compat with FirePrince's Unique Ascension Perks
		any_playable_country = {
			is_subject = yes
			has_overlord = root
			NOT = { is_in_federation_with = root }
		}
	}

	immediate = {
		every_playable_country = {
			limit = {
				is_subject = yes
				has_overlord = root
				NOR = {
					is_in_federation_with = root
					any_agreement = { agreement_preset = preset_auto_state }
				}
			}
			join_alliance = { who = root override_requirements = yes }
		}
	}
}

country_event = {
	id = vfefpulse.4000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_global_flag = civil_wars_active
		has_federation = yes

		any_subject = {
			any_agreement = { agreement_preset = preset_auto_state }
		}
	}

	immediate = {
		every_subject = {

				any_agreement = { agreement_preset = preset_auto_state }
			}
			leave_alliance = { override_requirements = yes }
		}
	}
}
