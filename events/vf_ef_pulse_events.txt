namespace = vfefpulse

country_event = {
	id = vfefpulse.1000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_federation = yes
		federation = { has_federation_type = vf_ef_federation }
		is_federation_leader = no
		is_subject_type = devstate
	}

	immediate = {
		leave_alliance = { override_requirements = yes }
	}
}

country_event = {
	id = vfefpulse.2000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_federation = yes
		federation = { has_federation_type = vf_ef_federation }
		is_federation_leader = no
		check_subject_join_wars = yes
	}

	immediate = {
		random_war = {
			limit = {
				any_war_participant = {
					is_overlord_to = root
					is_war_leader = yes
				}	
			}
			every_war_participant = {
				limit = { is_same_empire = root }
				prev = { remove_war_participant = root }
			}
		}
	}
}

country_event = {
	id = vfefpulse.9000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { has_global_flag = vf_ef_fix }
		has_federation = yes
		federation = {
			has_federation_type = vf_ef_federation
			NOT = { has_federation_law = allow_subjects_to_join_yes_empire }
		}
	}

	immediate = {
		federation = { set_allow_subjects_to_join = yes }
		set_global_flag = vf_ef_fix
	}
}
