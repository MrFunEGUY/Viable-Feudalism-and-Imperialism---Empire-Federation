# Feudal Empire: vassals can be anything
create_empire_fed_member_1 = {
	create_country = {
		name_list = random
		species = event_target:empire_planet_1_species
		type = default
		origin = "origin_vfef_empire_npc"
		ignore_initial_colony_error = yes
		day_zero_contact = no
		randomize_ethos = yes
		effect = {
			if = {
				limit = {
					has_civic = civic_fanatic_purifiers
				}
				force_remove_civic = civic_fanatic_purifiers
			}
			save_event_target_as = empire_member_1
			establish_communications_no_message = root
		}
	}
}
