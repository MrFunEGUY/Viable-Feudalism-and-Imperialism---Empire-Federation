# Triggers when the game starts
on_game_start = {
	events = {
		vfef_start.1000   			# Set Global Flag
	}
}

# Triggers when the game starts
# scope = country
on_game_start_country = {
	events = {
		vfef.5000   			# Empire Origin Event
	}
}


# This = subject
# From = subject's overlord
on_becoming_subject = {
	events = {
		vfef.1000				# Join Overlord's Federation if laws permit
		vfdiplo.1000			# Give empire fed subjects diplo weight modifier.
	}
}

# This = empire trying to leave federation
# From = empire who declined (federation leader)
on_ask_to_leave_federation_declined = {
	events = {
		vfef.2000
	}
}

# this = country
on_yearly_pulse_country = {
	events = {
		vfdiplo.1050			# remove diplo weight modifier if not a subject
		vfefpulse.1000
	}
}

# this = country, fired via action.220 from on_five_year_pulse_country
on_five_year_random_pulse_country = {
	random_events = {
		100 = vfef.3000			# Federation empire challenge
	}
}

on_war_beginning = {
	events = {
		vfefpulse.2000			# Remove correct subjects from wars
	}
}

# this = country
on_monthly_pulse_country = {
	events = {
		vfefpulse.9000			# Potential Bugfix
		vfefpulse.3000			# New vassal joins Empire
		vfefpulse.4000			# Autuonoms State leaves Empire
	}
}
