@cooldown = 7


invite_members_unanimous_vote_empire = {
	potential = { has_federation_type = vf_ef_federation }
	allow = {
		custom_tooltip = {
			NOT = {
				AND = {
					NOT = { has_federation_law = invite_members_unanimous_vote_empire }
					has_modifier = federation_law_category_invite_members_vote_cooldown
					has_any_federation_law_in_category = law_category_invite_members_vote_empire
				}
			}
		}
		custom_tooltip = {
			fail_text = EMPIRE_LAW_CHANGE
			NAND = {
				NOT = { has_federation_law = invite_members_unanimous_vote_empire }
				has_federation_flag = vf_law_changes_disabled
			}
		}
	}
	on_enact = {
		set_diplomacy_action_setting = {
			action = action_invite_to_federation
			settings = {
				vote_type = unanimous_vote
			}
		}
		if = {
			limit = { has_federation_flag = enable_federation_cooldowns }
			add_modifier = {
				modifier = federation_law_category_invite_members_vote_cooldown
				days = @cooldown
			}
		}
		add_cohesion = -5
	}

	ai_weight = {
		base = 200
		
		#Default support
		modifier = {
			desc = federation_acceptance_generally_positive
			add = 51
		}
	}
}
