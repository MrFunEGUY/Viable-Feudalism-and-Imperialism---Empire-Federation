# See wg_example.txt for an explanation

wg_independence = {
	casus_belli = cb_subject

	hide_if_no_cb = yes
	set_defender_wargoal = wg_assert_overlordship

	potential = {
		always = yes
	}
	possible = {
		custom_tooltip = {
			fail_text = cannot_use_against_emperor
			OR = {
				from = { is_galactic_emperor = no }
				is_galactic_community_member = no
			}
		}
		custom_tooltip = {
			fail_text = pax_galactica_active
			NAND = {
				is_active_resolution = resolution_emperor_pax_galactica
				is_galactic_community_member = yes
				from = { is_galactic_community_member = yes }
			}
		}
	}

	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}

	on_status_quo = {
		custom_tooltip = wg_independence_status_quo
		hidden_effect = {
			set_subject_of = {
				who = none
			}
		}
	}

	on_accept = {
		custom_tooltip = wg_independence_accept
		hidden_effect = {
			set_subject_of = {
				who = none
			}
			every_system_within_border = {
				limit = {
					from = { has_claim = prev }
				}
				remove_claims = { who = from }
			}
			from = {
				if = {
					limit = { is_variable_set = empire_fed_exp_before_war }
					clear_variable = empire_fed_exp_before_war
				}
			}
		}
	}

	on_wargoal_set = {
		every_system_within_border = {
			limit = {
				NOT = { from = { has_claim = prev } }
			}
			add_claims = { who = from }
		}
		if = {
			limit = {
				has_federation = yes
				federation = { has_federation_type = vf_ef_federation }
			}
			federation = {
				export_trigger_value_to_variable = {
					trigger = federation_experience
					variable = emp_fed_exp
				}
			}
			log = "WARGOAL"
			if = {
				limit = { federation = { num_members = 2 } }
				log = "LIMIT WORKING"
				from = { set_variable = { which = empire_fed_exp_before_war value = federation.emp_fed_exp } }
			}
			else = {
				federation = { # Reduce experience by 15%
					multiply_variable = { which = emp_fed_exp value = 0.85 }
					add_federation_experience = -50000
					add_federation_experience = emp_fed_exp
					clear_variable = emp_fed_exp
				}
			}
			leave_alliance = { override_requirements = yes }
		}
	}

	ai_weight = {
		weight = 100
	}
}
